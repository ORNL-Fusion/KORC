  subroutine korc_abort()
    !! @note Subroutine that terminates the simulation. @endnote
    INTEGER :: mpierr 
    !! MPI error status

    call MPI_BARRIER(MPI_COMM_WORLD,mpierr)

    call MPI_ABORT(MPI_COMM_WORLD, -2000, mpierr)
  end subroutine korc_abort
